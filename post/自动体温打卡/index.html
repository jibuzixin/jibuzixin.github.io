体温自动打卡脚本

原改于我们王导

Posted by 忌不自信 on Wednesday, November 2, 2022

通过手机抓包软件抓包，然后看一下是哪个请求去查信息，打卡的，怎么打的，然后用python模拟发送请求就行了，都没加密，一下子就搞定了。 下面就是代码，就是发送请求，获取请求，原理还是挺简单的。

然后再写一个shell脚本，定时一下，其实也可以不用写shell，但后面要做的话就只用改shell就行了，其实更方便点吧

import random
from re import L
import requests
import json
import time
import xlrd

import datetime
from time import strftime

from urllib import parse


# 根据身份证号执行一个人的打卡动作
def daka(identity):
    head = {
        'Content-Type': 'application/x-www-form-urlencoded; charset=utf-8',
        'Accept-Encoding': 'gzip'
    }
    # 定义服务器返回消息的变量
    global msg
    #定义计数打卡的总人数
    global count
    msg = ''
    # 随机体温
    temp = random.randint(360, 372) / 10
    print(f"体温:{temp}")
    # 这个函数是要转换成为十六进制
    # pos = parse.quote(pos)
    try:
        print("开始请求打卡")
        r = requests.post(
            url='http://yx.ty-ke.com:80/Home/Monitor/monitor_add',
            headers=head,
            data=f'mobile={identity}&title={temp}&jk_type=%E5%81%A5%E5%BA%B7&wc_type=%E5%90%A6&jc_type=%E5%90%A6&province={parse.quote(province)}&city={parse.quote(city)}&district={parse.quote(district)}&address={parse.quote(address)}&is_verify=0&health_code=&shibie_type=&travel_code='
        )
        print("请求完成，正在处理数据... ...")
        # 现在要通过字符串创建map，检查结果
        msg = json.loads(r.text)['msg']
        print("打卡成功！")
        count += 1
    except:
        print('准备打卡的时候请求错误')


# 根据身份证号，获取最近一次打卡的地址信息
def getAddrInfo(identity):
    get_addr_head = {
        'Content-Type': 'application/x-www-form-urlencoded; charset=utf-8',
        'Accept-Encoding': 'gzip'
    }
    try:
        r = requests.post(
            url='http://yx.ty-ke.com:80/Home/Monitor/monitor_daka',
            headers=get_addr_head,
            data=f'mobile={identity}'
        )
        # 现在要通过字符串创建map，检查结果
        res_map = json.loads(r.text)['data']
    except:
        print('地址信息请求错误')


    # print(res_map)
    # 定义地址变量
    global province
    province = ''
    global city
    city = ''
    global address
    address = ''
    global district
    district = ''
    global id
    id = ''

    for pos in res_map:
        try:
            if pos['id'] != None:
                # id不空的时候一定有地址信息，获取地址信息
                province = pos['province']
                city = pos['city']
                address = pos['address']
                district = pos['district']
                id = pos['id']
                break
        except:
            continue
